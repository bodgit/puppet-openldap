function openldap::flatten_unique(Optional[Array[OpenLDAP::Unique, 1]] $values) {

  $values ? {
    undef   => undef,
    default => $values.map |OpenLDAP::Unique $value| {
      join(delete_undef_values([
        $value['strict'] ? {
          true    => 'strict',
          default => undef,
        },
        $value['ignore'] ? {
          true    => 'ignore',
          default => undef,
        },
        join($value['uri'], ' '),
      ]), ' ')
    }
  }
}
