function openldap::flatten_indices(Optional[Array[OpenLDAP::Index, 1]] $values) {

  $values ? {
    undef   => undef,
    default => $values.map |OpenLDAP::Index $value| {
      size($value) ? {
        1       => "${join($value[0], ',')}",
        default => "${join($value[0], ',')} ${join($value[1], ',')}",
      }
    }
  }
}
